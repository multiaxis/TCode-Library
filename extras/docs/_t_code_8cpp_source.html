41void TCode::inputChar(const char input)
42{
43 bool success = inputBuffer.push(input); // pushes the inputted char to the buffer
44 if (!success && inputBuffer.isFull()) // if the buffer is full then execute the first command and then push the char to the buffer
45 {
46 executeNextBufferCommand();
47 inputBuffer.push(input);
48 }
49
50 if (input == '\n') // if a newline is encountered run all commands till the buffer is empty
51 {
52 while (!inputBuffer.isEmpty())
53 executeNextBufferCommand();
54 inputBuffer.clear();
55 }
56}

206void TCode::executeNextBufferCommand()
207{
208 unsigned char command[MAX_COMMAND_BUFFER_LENGTH_COUNT] = {'\0'};
209 size_t index = 0;
210 int blevel = 0;
211 while (!inputBuffer.isEmpty() && (index < MAX_COMMAND_BUFFER_LENGTH_COUNT - 1))
212 {
213 if (inputBuffer.peek() == '[')
214 blevel += 1;
215 if (inputBuffer.peek() == ']')
216 blevel -= 1;
217
218 if (inputBuffer.peek() == ' ' && (blevel == 0))
219 {
220 inputBuffer.pop();
221 break;
222 }
223
224 if (inputBuffer.peek() == '\n')
225 {
226 inputBuffer.pop();
227 break;
228 }
229
230 if (blevel < 0)
231 break;
232
233 command[index++] = inputBuffer.pop();
234 }
235
236 readCommand(command, index + 1);
237}

239void TCode::readCommand(unsigned char *command, size_t length)
240{
241 TCode_Command_Type type = TCodeParser::getCommandType(command, length, 0); // find what command was read
242 // Switch between command types
243 switch (type) // depending on the command type found parse the inputted command and execute the correct command function
244 {
245 case TCode_Command_Type::Axis:
246 {
247 TCode_Axis_Command result;
248 if (TCodeParser::parseAxisCommand(command, length, result))
249 runAxisCommand(result);
250 break;
251 }
252 case TCode_Command_Type::Device:
253 {
254 TCode_Device_Command result;
255 if (TCodeParser::parseDeviceCommand(command, length, result))
256 runDeviceCommand(result);
257 break;
258 }
259 case TCode_Command_Type::Setup:
260 {
261 TCode_Setup_Command result;
262 if (TCodeParser::parseSetupCommand(command, length, result))
263 runSetupCommand(result);
264 break;
265 }
266 case TCode_Command_Type::External:
267 {
268 TCode_External_Command result;
269 if (TCodeParser::parseExternalCommand(command, length, result))
270 runExternalCommand(result);
271 break;
272 }
273 default:
274 break;
275 }
276}

352void TCode::printSavedAxisValues()
353{

355 {
356 TCodeAxis *temp = nullptr;
357 if (axisBuffer.get(i, temp))
358 {
359 String str_id;
360 TCodeParser::getStrfromID(temp->getChannelID(), str_id);
361 String name = "AXIS-MIN-";
362 name += str_id;
363 unsigned int min = 0;
364 if (!settingManager.getSetting(name.c_str(), min))
365 {
366 settingManager.setSetting(name.c_str(), 0);
367 }
368
369 name = "AXIS-MAX-";
370 name += str_id;
371 unsigned int max = TCODE_MAX_AXIS;
372 if (!settingManager.getSetting(name.c_str(), max))
373 {
374 settingManager.setSetting(name.c_str(), TCODE_MAX_AXIS);
375 }
376
377 name = "";
378 name += str_id;
379 name += '-';
380 name += temp->getName();
381 name += '-';
382 name += String(min);
383 name += '-';
384 name += String(max);
385
386 println(name);
