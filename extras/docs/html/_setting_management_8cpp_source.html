61bool Settings::hasSetting(const char *setting)

105bool Settings::getFile(String &out)
106{
107 if (!isMounted())
108 return false;
109 out = "";
110 File file = SPIFFS.open(filepath);
111
112 if (!file)
113 {
114#ifdef DEBUG
115 Serial.print(F("SM: Could not find file \""));
116 Serial.print(filepath);
117 Serial.println(F("\""));
118#endif
119 return false;
120 }
121
122 while (file.available())
123 {
124 uint8_t current_byte = file.read();
125 if (current_byte == '\n')
126 continue;
127 out += (char)current_byte;
128 if (out.length() >= JSON_FILE_SIZE)
129 {
130#ifdef DEBUG
131 Serial.print(F("SM: File larger than "));
132 Serial.println(JSON_FILE_SIZE);
133#endif
134 file.close();
135 return false;
136 }
137 }
138 file.close();
139 return true;
140}
