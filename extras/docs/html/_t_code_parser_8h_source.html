95 static long getNextInt(unsigned char *buffer, const size_t length, size_t &index)
96 {
97 size_t count = 0;
98 long accum = 0;
99 while (isnumber(*(buffer + index))) // while there is a number at the index we are at in the string
100 {
101 accum *= 10; // multiply the accumulator first to get the correct output value
102 accum += static_cast<long>(toupper(*(buffer + index)) - '0'); // get next int value '0' - '9' subtracting '0' gets the integer value of the next unit
103 index++;
104 count++; // increase the indeces count to count the digits
105 }
106
107 if (count == 0) // no chars were found with 0 - 9
108 return -1;
109
110 while (count < 4) // if less than 4 digits were found make up for it by multiplying eg 1 = 1000, 90 = 9000, 001 = 10
111 {
112 accum *= 10;
113 count++;
114 }
115
116 return accum;
117 }

141 static size_t uintToStr(unsigned long value, unsigned char *buffer, const size_t length, size_t &placeIndex)
142 {
143 const int base = 10;
144 char buf[30] = {'\0'};
145 int i = 30;
146
147 if (value == 0)

156 }
157
158 unsigned char *buffer_location = buffer + placeIndex;
159 size_t count = 0;

162 if ((count + placeIndex) >= length - 1)

164 break;
165 }
166
167 *buffer_location = buf[i];
168 buffer_location++;
169 }
