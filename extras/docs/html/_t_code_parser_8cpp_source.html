9bool TCodeParser::idValidRange(const TCode_Channel_Type type, const uint8_t channel, const uint8_t maxChannel)
10{
11 uint8_t tempMaxChannel = maxChannel;
12 if (maxChannel >= TCODE_MAX_CHANNEL_COUNT)
13 tempMaxChannel = TCODE_MAX_CHANNEL_COUNT - 1;
14 if (maxChannel < 0)
15 tempMaxChannel = 0;
16
17 if ((channel > tempMaxChannel) || (channel < 0)) // check if channel number is not out of range
18 return false;
19
20 switch (type) // check if the type is correct
21 {
22 case TCode_Channel_Type::Auxiliary:
23 case TCode_Channel_Type::Linear:
24 case TCode_Channel_Type::Rotation:
25 case TCode_Channel_Type::Vibration:
26 break;
27 default:
28 return false;
29 }
30
31 return true;
32}

49int TCodeParser::getStrfromID(unsigned char *buffer, const size_t length, const TCode_ChannelID &id)
50{
51 size_t index = 0;
52 switch (id.type) // gets the Channel type char and appends to the output string
53 {
54 case TCode_Channel_Type::Auxiliary:
55 *(buffer + index) = 'A';

57 case TCode_Channel_Type::Linear:
58 *(buffer + index) = 'L';

60 case TCode_Channel_Type::Rotation:

62 break;
63 case TCode_Channel_Type::Vibration:
64 *(buffer + index) = 'V';
65 break;
66 default:
67 *(buffer + index) = '?';
68 }
69 index++;
70 index += uintToStr(id.channel, buffer, length, index);
71

107TCode_Axis_Ramp_Type TCodeParser::getRampTypeFromStr(unsigned char *buffer, const size_t length, size_t &startIndex)
108{
109 TCode_Axis_Ramp_Type rampType = TCode_Axis_Ramp_Type::None;
110 char first = toupper(getCharAt(buffer, length, startIndex));
111 char second = toupper(getCharAt(buffer, length, startIndex + 1));
112 switch (first) // Decode what Ramp type it is
113 {
114 case '<':
115 if (second == '>')
116 {
117 startIndex += 2;
118 rampType = TCode_Axis_Ramp_Type::EaseInOut;
119 }
120 else
121 {
122 startIndex++;
123 rampType = TCode_Axis_Ramp_Type::EaseIn;
124 }
125 break;
126 case '>':
127 {
128 startIndex++;
129 rampType = TCode_Axis_Ramp_Type::EaseOut;
130 }
131 break;
132 case '=':
133 {
134 startIndex++;
135 rampType = TCode_Axis_Ramp_Type::Linear;
136 }
137 break;
138 }
139 return rampType;
140}

190bool TCodeParser::parseAxisCommand(unsigned char *buffer, const size_t length, TCode_Axis_Command &out)
191{
192 bool valid = true;
193 size_t index = 0;
194 TCode_ChannelID id = getIDFromStr(buffer, length, index);
195 TCode_Axis_Extention_Type extentionType = TCode_Axis_Extention_Type::None;
196 TCode_Axis_Ramp_Type rampType = TCode_Axis_Ramp_Type::Linear;
197 long command_value = 0;
198 long commandExtention = 0;
199
200 if (id.type == TCode_Channel_Type::None)
201 valid = false;
202
203 command_value = getNextInt(buffer, length, index);
204 if (command_value == -1)
205 valid = false;
206 command_value = constrain(command_value, 0, TCODE_MAX_AXIS);
207
208 if (isextention(getCharAt(buffer, length, index)))
209 {
210 extentionType = getExtentionTypeFromStr(buffer, length, index);
211 commandExtention = getNextInt(buffer, length, index);
212 if (commandExtention == -1)
213 valid = false;
214 commandExtention = constrain(commandExtention, 0, TCODE_MAX_AXIS_MAGNITUDE);
215 }
216
217 if (isramp(getCharAt(buffer, length, index)))
218 {
219 rampType = getRampTypeFromStr(buffer, length, index);
220 }
221
222 if (toupper(getCharAt(buffer, length, index)) != '\0') // if the command has been processed and there are still characters left over the command has not been processed correctly/the command is incorrect
223 valid = false;
224 if (!idValid(id)) // make sure that the ID is valid if it isnt then the command is not valid
225 valid = false;
226
227 TCode_Axis_Data data;
228 data.commandValue = command_value;
229 data.commandExtention = commandExtention;
230 data.extentionType = extentionType;
231 data.rampType = rampType;
232
233 out.Data = data;
234 out.ID = id;
235
236 return valid;
237}

239bool TCodeParser::parseSetupCommand(unsigned char *buffer, const size_t length, TCode_Setup_Command &out)
240{
241 bool valid = true;
242 size_t index = 0;
243 if (toupper(getCharAt(buffer, length, index++)) != '$')
244 valid = false;
245
246 TCode_ChannelID id = getIDFromStr(buffer, length, index); // get the ID
247 if (toupper(getCharAt(buffer, length, index++)) != '-')
248 valid = false;
249
250 long minValue = getNextInt(buffer, length, index); // Get the first minimum value
251
252 if (toupper(getCharAt(buffer, length, index++)) != '-')
253 valid = false;
254
255 long maxValue = getNextInt(buffer, length, index); // Get the seccond maximum value
256 if ((toupper(getCharAt(buffer, length, index)) != '\0') || (minValue == -1) || (maxValue == -1) || (!idValid(id)))
257 valid = false;
258 minValue = constrain(minValue, 0, TCODE_MAX_AXIS); // constrain the values to the maximum axis constraint
259 maxValue = constrain(maxValue, 0, TCODE_MAX_AXIS);
260 if (minValue > maxValue) // if the minimum is larger than the maximum the command is not valid
261 valid = false;
262
263 TCode_Save_Entry values = {static_cast<int>(minValue), static_cast<int>(maxValue)};
264 out = {id, values};
265 return valid;
266}
