94Q16Fixed lerpQ16(Q16Fixed start, Q16Fixed stop, Q16Fixed t)
95{
96 // lerp is defined as mixed = ((1.0 - t) * a) + (t * b)
97 t = constrainQ16(t, 0, Q16fromInt(1));
98 Q16Fixed tn = subQ16(Q16fromInt(1), t);
99 Q16Fixed a = multQ16(tn, start);
100 Q16Fixed b = multQ16(t, stop);
101 return addQ16(a, b);
102}

119long mapQ16(long in, long inStart, long inEnd, long outStart, long outEnd)
120{
121 Q16Fixed t = Q16fromInt(in);
122 t = constrainQ16(t, Q16fromInt(inStart), Q16fromInt(inEnd));
123 t = t - Q16fromInt(inStart); // Find the distance of the in value from inStart
124 t = divQ16(t, subQ16(Q16fromInt(inEnd), Q16fromInt(inStart))); // Find where the inValue lies within the range between inStart and inEnd should be (0.0 - 1.0) for values where in is in the range specified
125 t = constrainQ16(t, 0, Q16fromInt(1)); // constrain the value from 0.0 to 1.0
126 t = multQ16(t, subQ16(Q16fromInt(outEnd), Q16fromInt(outStart))); // multiply the value of how far along the range between outStart and outEnd
127 t = addQ16(t, Q16fromInt(outStart)); // Get the value to lie between outStart and outEnd
128 t = constrainQ16(t, Q16fromInt(outStart), Q16fromInt(outEnd)); // Constrain the value so it does not go out of bounds
129 t = IntfromQ16(t);
130 return t; // Convert the value back to an int
131}
