109inline bool Settings::getSetting(const char *setting, T &settingValue)
110{
111
112#ifdef DEBUG
113 Serial.print(F("SM: getting setting "));
114 Serial.println(setting);
115#endif
116
117 StaticJsonDocument<JSON_FILE_SIZE> doc;
118 String file_data;
119 if (!getFile(file_data))
120 {
121#ifdef DEBUG
122 Serial.println(F("SM: error getting file data "));
123#endif
124 return false;
125 }
126
127 DeserializationError error = deserializeJson(doc, file_data);
128 if (error)
129 {
130#ifdef DEBUG
131 Serial.print(F("SM: deserializeJson() failed: "));
132 Serial.println(error.f_str());
133#endif
134 return false;
135 }
136
137 #ifdef DEBUG
138 Serial.print(F("SM: keys "));
139 JsonObject documentRoot = doc.as<JsonObject>();
140 for(JsonPair keyValue : documentRoot) {
141 Serial.print(F("\""));
142 Serial.print(keyValue.key().c_str());
143 Serial.print("\":\"");
144 Serial.print(keyValue.value().as<String>());
145 Serial.println(F("\""));
146 }
147 #endif
148
149
150
151
152 if (!doc.containsKey(setting))
153 {
154 #ifdef DEBUG
155 Serial.println(F("SM: key not found"));
156 #endif
157 return false;
158 }
159
160 settingValue = doc[setting].as<T>();
161
162 return true;
163}

168

177 String fileData;
178 if (!getFile(fileData))
179 {
180 return false;

184 if (error)
185 {
186 Serial.print(F("SM: deserializeJson() failed: "));
187 Serial.println(error.f_str());
188 return false;
189 }
190
191 doc[setting] = settingValue;
192 String out;
193 serializeJson(doc, out);
194 writeFile(out);
195 return true;
