19unsigned long TCodeAxis::getDeltaTime(int start_position, int target_value, TCode_Axis_Extention_Type extentionType, long extentionValue, TCode_Axis_Ramp_Type rampType)
20{
21 unsigned long delta_time = 0;
22
23 switch (extentionType)
24 {
25 case TCode_Axis_Extention_Type::Speed:
26 {
27 delta_time = abs(target_value - start_position);
28 delta_time *= 100;

34 break;

42 }

47void TCodeAxis::set(int target_value, TCode_Axis_Extention_Type extentionType, long extentionValue, TCode_Axis_Ramp_Type rampType)
48{
49 unsigned long t = getLastMillis(); // This is for adding this state
50 target_value = constrain(target_value, 0, TCODE_MAX_AXIS);
51 extentionValue = constrain(extentionValue, 0, TCODE_MAX_AXIS_MAGNITUDE);
52
53 int last_target = getLastTargetPosition();
54 unsigned long delta_time = getDeltaTime(last_target, target_value, extentionType, extentionValue, rampType);
55
56 TCode_Axis_State new_state;
57 new_state.startTime = t;
58 new_state.endTime = t + delta_time;
59 new_state.startValue = last_target;
60 new_state.endValue = target_value;
61 new_state.rampType = rampType;
62 dataQueue.push(new_state);
63
64 int lastInterval = millis() - lastCommandTime;
65 int difference = abs(lastInterval - minInterval);
66 if ((lastInterval > minInterval) && (minInterval < TCODE_MAX_AXIS_SMOOTH_INTERVAL))
67 {
68 minInterval += difference / 2;
69 }
70 else if ((lastInterval < minInterval) && (minInterval > TCODE_MIN_AXIS_SMOOTH_INTERVAL))
71 {
72 minInterval -= difference / 2;
73 }
74 minInterval = constrain(minInterval, TCODE_MIN_AXIS_SMOOTH_INTERVAL, TCODE_MAX_AXIS_SMOOTH_INTERVAL);
75 lastCommandTime = millis();
76}

131int TCodeAxis::getPositionMapped(TCode_Axis_State &state, bool &inRange)
132{
133 int x; // This is the current axis position, 0-9999
134 unsigned long t = millis();
135 inRange = true;
136 if (t >= state.endTime)
137 {
138 inRange = false;
139 return state.endValue;
140 }
141
142 if (t < state.startTime)
143 {
144 inRange = false;
145 return state.startValue;
146 }
147
148 switch (state.rampType)
149 {
150 case TCode_Axis_Ramp_Type::Linear:
151 x = mapQ16(t, state.startTime, state.endTime, state.startValue, state.endValue);
152 break;
153 case TCode_Axis_Ramp_Type::EaseIn:
154 x = mapEaseIn(t, state.startTime, state.endTime, state.startValue, state.endValue);
155 break;
156 case TCode_Axis_Ramp_Type::EaseOut:
157 x = mapEaseOut(t, state.startTime, state.endTime, state.startValue, state.endValue);
158 break;
159 case TCode_Axis_Ramp_Type::EaseInOut:
160 x = mapEaseInOut(t, state.startTime, state.endTime, state.startValue, state.endValue);
161 break;
162 default:
163 x = mapQ16(t, state.startTime, state.endTime, state.startValue, state.endValue);
164 }
165
166 if (x > TCODE_MAX_AXIS)
167 x = TCODE_MAX_AXIS;
168 if (x < 0)
169 x = 0;
170
171 return x;
172}
